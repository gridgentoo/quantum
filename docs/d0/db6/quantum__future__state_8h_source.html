130 void ThrowFutureException(FutureState state)
131 {
132  switch (state)
133  {
134  case FutureState::PromiseNotSatisfied: throw PromiseNotSatisfiedException();
135  case FutureState::PromiseAlreadySatisfied: throw PromiseAlreadySatisfiedException();
136  case FutureState::BrokenPromise: throw BrokenPromiseException();
137  case FutureState::FutureAlreadyRetrieved: throw FutureAlreadyRetrievedException();
138  case FutureState::NoState: throw NoStateException();
139  case FutureState::BufferingData: throw BufferingDataException();
140  case FutureState::BufferClosed: throw BufferClosedException();
141  default: throw std::logic_error("Invalid future state");
142  }
